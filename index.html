-- Pevolution Secure Loader
local function loadSecure()
    local gateway = "pevolution.vercel.app"
    local success, script = pcall(function()
        return game:HttpGet(
            "https://" .. gateway .. "/api/script",
            true
        )
    end)
    
    if success then
        local func = loadstring(script)
        if func then func() end
    end
end

loadSecure()
